def getPresidents(url):
    urlreq = requests.get(url)
    urltext = urlreq.text
    urllines = re.findall(r'(<big><a href="(.*?)")',urltext)
    urls = []
    for i in range(len(urllines)):
        urls.append(urllines[i][1])
    url_presidents = [wiki + v for v in urls]
    return url_presidents

def getBirthdays(url_presidents):
    bday = []
    name = []
    for ele in range(len(url_presidents)):
        urlreq = requests.get(url_presidents[ele])
        urltext = urlreq.text
        urllines = re.findall(r'<span class="bday">(.*?)</span>\)',urltext)
        urlname = re.findall(r'"wgTitle":"(.*?)"',urltext)
        bday.append(dt.strptime(urllines[0], '%Y-%m-%d'))
        name.append(urlname[0])
    ser = pd.Series(name, index=bday)
    return ser

import numpy as np
import time
import math
import matplotlib.pyplot as plt
#% matplotlib inline
np.random.seed(0)

def minor2(A,i,j):
    A=np.delete(A,(i),axis=0)
    A=np.delete(A,(j),axis=1)
    return A


def det(A):
    d=0
    if A.shape==(2,2):
        return A[1,1]*A[0,0]-A[0,1]*A[1,0]
    else:
        for i in range(A.shape[0]):
            B=minor2(A,0,i)
            d+=math.pow(-1,i)*A[0][i]*det(B)
    return d

According to the graphs, Numpy functions take lesser time for sorting, multiplication and computation of determinant. The range in the question takes a lot of time for computation, so, I have taken a smaller range for the experiment. For Multiplication and determinant calculation, time required for Numpy ufuncs is always constant while for Matrix Multiplication it varies as per the size of the matrix. The time required varies linearly and exponentially, respectively, for sorting and determinant calculation, and matrix multiplication.
